            url: 'https://ptx.transportdata.tw/MOTC/v2/Rail/TRA/LiveBoard?$format=JSON',
            format: 'JSON',
            success: function (data) {
                let tbl = $('#tbl');
                let context="";
                let tripLine, tripLineStr;
                let delayTime, delayTimeStr;
                let direction;
                tbl.append("站名車次車種山/海線到站時間發車時間誤點時間");
                for (let i = 0; i < data.length; i++) {
                    context = "";
                    tripLine = data[i].TripLine;
                    tripLineStr = "";                                        
                    if (tripLine == 0) {
                        tripLineStr = "不經山海線";
                    } else if (tripLine==1) {
                        tripLineStr = "山線";
                    } else if (tripLine==2) {
                        tripLineStr = "海線";
                    }
                    direction = data[i].Direction;
                    if (direction == 0) {
                        tripLineStr += "(順行)";
                    } else if (direction == 1) {
                        tripLineStr += "(逆行)";
                    }                    
                    delayTime = data[i].DelayTime;
                    delayTimeStr = "";
                    if (delayTime == 0) {
                        delayTimeStr = "準點";
                    } else {
                        delayTimeStr = "" + delayTime+"分" + "";
                    }

                    context = "" + data[i].StationName.Zh_tw + "" + data[i].TrainNo + "" + data[i].TrainTypeName.Zh_tw + "" + tripLineStr + ">" + data[i].ScheduledArrivalTime + "" + data[i].ScheduledDepartureTime + "" + delayTimeStr + "";
                    tbl.append(context);                                  
                    tbl.append("");
                }
            },
            error: function () {
                alert('Ajax Fail!!');
            }
        });
